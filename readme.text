3. intall tailwind => 
    npm install -D tailwindcss postcss autoprefixer
    npx tailwindcss init -p
    => replace config with
    /** @type {import('tailwindcss').Config} */
export default {
  content: [
    "./index.html",
    "./src/**/*.{js,ts,jsx,tsx}",
     ],
      theme: {
         extend: {},
       },
      plugins: [],
    }
    => add this to index.css
    @tailwind base;
    @tailwind components;
    @tailwind utilities;
4. npm run dev
